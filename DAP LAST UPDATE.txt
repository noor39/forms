-- Create tables
CREATE TABLE users (
    user_id NUMBER(20)PRIMARY KEY,
    user_name VARCHAR2(50) NOT NULL,
    password VARCHAR2(100) NOT NULL,
    email VARCHAR2(100) UNIQUE,
    phone NUMBER(15) NOT NULL,
    role VARCHAR2(20) NOT NULL CHECK (role IN ('admin', 'doctor', 'staff', 'patient'))
);

CREATE TABLE Doctors (
    doctor_id NUMBER(20) PRIMARY KEY,
    user_id NUMBER(20) REFERENCES users(user_id),
    name VARCHAR2(100) NOT NULL,
    qualification VARCHAR2(100),
    designation VARCHAR2(100),
    specialization VARCHAR2(100),
    phone NUMBER(15) NOT NULL,
    email VARCHAR2(100)
);

CREATE TABLE Schedules (
    schedule_id NUMBER(20) PRIMARY KEY,
    doctor_id NUMBER(20) REFERENCES doctors(doctor_id),
    schedule_date date,
    schedule_time TIMESTAMP Not null,
    status VARCHAR2(30)
);

CREATE TABLE Patients (
    patient_id number(30) PRIMARY KEY,
    phone NUMBER(15) NOT NULL,
    name VARCHAR2(100) NOT NULL,
    Date_of_Birth DATE,
    Age NUMBER(3), -- Updated from VARCHAR2 to NUMBER
    Weight DECIMAL(5,2),
    gender VARCHAR2(10) CHECK (gender IN ('Male', 'Female', 'Other')),
    email VARCHAR2(100),
    address CLOB,
    create_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    created_by VARCHAR2(15)
);


CREATE TABLE Appointments (
    appointment_id NUMBER(20) PRIMARY KEY,
    serial_no NUMBER,
    patient_id number(30) REFERENCES patients(patient_id),
    appointment_datetime TIMESTAMP NOT NULL,
    status VARCHAR2(20) DEFAULT 'pending' 
        CHECK (status IN ('pending', 'confirmed', 'completed', 'cancelled')),
    schedule_id number(20)  REFERENCES Schedules(schedule_id),
    create_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    created_by VARCHAR2(15)
);

CREATE TABLE Generic (
    generic_id number(20)  PRIMARY KEY ,
    generic_name VARCHAR2(100) NOT NULL
);

CREATE TABLE Medicine (
    medicine_id number(10)PRIMARY KEY ,
    brand_name VARCHAR(100) NOT NULL,
    generic_id number(20) REFERENCES Generic(generic_id),
    dosage_form varchar2(25),
    strength varchar2(50)  
);


CREATE TABLE Prescription (
    prescription_id NUMBER(10) PRIMARY KEY,
    appointment_id NUMBER(20) REFERENCES Appointments(appointment_id),
    doctor_id NUMBER(20) REFERENCES Doctors(doctor_id),
    issue_date DATE NOT NULL,
    test_discount NUMBER(5,2) DEFAULT 0,
    create_date DATE DEFAULT SYSDATE,
    created_by VARCHAR2(30)
);

CREATE TABLE Prescription_Medicine (
    prescription_medicine_id number(20) PRIMARY KEY ,
    prescription_id NUMBER(10) references prescription(prescription_id),
    medicine_id number(10) references medicine(medicine_id),
    dosage varchar2(100),
    duration VARCHAR2(50)
);

CREATE TABLE Test_List (
    test_id number(20) PRIMARY KEY,
    test_name VARCHAR2(100) NOT NULL,
    test_category VARCHAR2(100)
);

CREATE TABLE Prescribed_Test (
    prescribed_test_id number(20) PRIMARY KEY,
    test_id number(20) references test_list(test_id),
    prescription_id number(10) references prescription(prescription_id),
    test_name VARCHAR2(100)
);

CREATE TABLE Test_Report_Master (
    test_report_master_id number PRIMARY KEY,
    doctor_id number(15) REFERENCES doctors(doctor_id),
    sample_date DATE,
    patient_id number(30) REFERENCES patients(patient_id),
    report_date DATE,
    test_id number(20) references test_list(test_id),
    report_by VARCHAR(100),
    created_by varchar2(30)
);

CREATE TABLE Test_Report_Details (
    test_report_details_id number PRIMARY KEY,
    test_report_master_id number references test_report_master(test_report_master_id),
    particulars VARCHAR2(100),
    reference_value VARCHAR2(100),
    result_value VARCHAR2(100)
);

CREATE TABLE Investigation (
    investigation_id number(20) PRIMARY KEY,
    prescription_id NUMBER(30) REFERENCES prescription(prescription_id),
    problem_name VARCHAR2(100)
);

CREATE TABLE Advice (
    advice_id number(20) PRIMARY KEY,
    prescription_id number(10) references prescription(prescription_id),
    advice_type VARCHAR2(50),
    advice_text varchar2(100),
    follow_up_day date
);

CREATE TABLE Expense_Master (
    expense_id number(20) PRIMARY KEY,
    user_id NUMBER(20) REFERENCES users(user_id),
    expense_date DATE,
    category VARCHAR2(100),
    total_amount number(10,2)
);

CREATE TABLE Expense_Details (
    detail_id number PRIMARY KEY,
    expense_id number(20) references expense_master(expense_id),
    description varchar2(100),
    amount number(10,2)
);


-- TRIGGER trg_expense_details_aiud TOTAL AMOUNT AUTO UPDATE
CREATE OR REPLACE TRIGGER trg_expense_details_aiud
FOR INSERT OR DELETE OR UPDATE ON Expense_Details
COMPOUND TRIGGER

    TYPE expense_id_tab IS TABLE OF VARCHAR2(10);
    v_expense_ids expense_id_tab := expense_id_tab();

    AFTER EACH ROW IS
    BEGIN
        -- Insert/Update case
        IF INSERTING OR UPDATING THEN
            v_expense_ids.EXTEND;
            v_expense_ids(v_expense_ids.COUNT) := :NEW.expense_id;
        END IF;

        -- Delete case
        IF DELETING THEN
            v_expense_ids.EXTEND;
            v_expense_ids(v_expense_ids.COUNT) := :OLD.expense_id;
        END IF;
    END AFTER EACH ROW;

    AFTER STATEMENT IS
    BEGIN
        FOR i IN 1 .. v_expense_ids.COUNT LOOP
            DECLARE
                v_total NUMBER(10,2);
            BEGIN
                SELECT NVL(SUM(amount), 0)
                INTO v_total
                FROM Expense_Details
                WHERE expense_id = v_expense_ids(i);

                UPDATE Expense_Master
                SET total_amount = v_total
                WHERE expense_id = v_expense_ids(i);
            END;
        END LOOP;
    END AFTER STATEMENT;

END trg_expense_details_aiud;
/

--create trigger auto age insert
CREATE OR REPLACE TRIGGER trg_set_patient_age
BEFORE INSERT OR UPDATE ON Patients
FOR EACH ROW
WHEN (NEW.Date_of_Birth IS NOT NULL)
BEGIN
    :NEW.Age := TRUNC(MONTHS_BETWEEN(SYSDATE, :NEW.Date_of_Birth) / 12);
END;
/

--Appointments auto create_date
CREATE OR REPLACE TRIGGER trg_appointment_audit
BEFORE INSERT ON Appointments
FOR EACH ROW
BEGIN
    :NEW.create_date := SYSTIMESTAMP;
END;
/

--TRIGGER auto trg_appointment_schedule_update
CREATE OR REPLACE TRIGGER trg_appointment_update
AFTER INSERT OR UPDATE OF status ON Appointments
FOR EACH ROW
DECLARE
    v_schedule_status Schedules.status%TYPE;
BEGIN
    IF :NEW.status = 'confirmed' AND (:OLD.status IS NULL OR :OLD.status != 'confirmed') THEN
        UPDATE Schedules
        SET status = 'Booked'
        WHERE schedule_id = :NEW.schedule_id;
      ELSIF :NEW.status = 'cancelled' AND (:OLD.status IS NULL OR :OLD.status != 'cancelled') THEN
        UPDATE Schedules
        SET status = 'Available'
        WHERE schedule_id = :NEW.schedule_id;
    END IF;
END;
/
